# ENTERPRISE DEPENDENCY MANAGEMENT:
#   All versions are PINNED (==). Never use >= or ~= in production containers.
#   Reason: A minor version upgrade to a dependency could silently change model
#   prediction behavior, API response format, or introduce security vulnerabilities.
#
#   PROCESS:
#     1. Develop with pinned versions
#     2. Periodically audit with: pip-audit (security) + pip list --outdated
#     3. Upgrade in a branch, run full test suite, then promote
#     4. Use dependabot / Renovate Bot for automated PR creation on updates
#
#   SBOM (Software Bill of Materials):
#     In enterprise, a requirements.txt with pinned versions is the foundation
#     of your SBOM — a legal/security requirement in regulated industries.
#     Tools: syft, cyclonedx-py generate full SBOM from requirements.txt.

# ── Web Framework ──────────────────────────────────────────────────────────────
# FastAPI: async web framework. Used by Netflix, Uber, Microsoft for ML serving.
# Industry standard for Python ML APIs (over Flask) because:
#   - Native async/await support (crucial for high-concurrency inference)
#   - Automatic OpenAPI/Swagger documentation from Pydantic schemas
#   - Built-in request validation (Pydantic integration)
#   - Performance: comparable to NodeJS, Go in async benchmarks
fastapi==0.115.0

# Uvicorn: ASGI server that runs FastAPI
# ASGI = Asynchronous Server Gateway Interface (async-capable, unlike WSGI/Gunicorn)
# In Kubernetes: uvicorn runs inside the container
#                workers=1 per pod (K8s manages parallelism via pod replicas, not workers)
# --workers > 1 in prod: use only if you need multiple CPU-bound workers per pod
#              (trade-off: multiple workers = multiple model copies in RAM)
uvicorn[standard]==0.30.6

# ── ML & Numerical ────────────────────────────────────────────────────────────
# scikit-learn: The model training and inference library
# Version MUST match exactly what was used during training.
# Mismatched sklearn version → model.pkl may fail to deserialize or behave differently.
# Enterprise rule: sklearn version in inference container == sklearn version in training job
scikit-learn==1.5.2
numpy==1.26.4
pandas==2.2.3       # For data exploration in train_and_save.py (not needed at inference)

# joblib: sklearn's preferred serialization library (included with sklearn but explicit)
# Handles memory-mapped numpy arrays efficiently for large model files
joblib==1.4.2

# ── Observability ─────────────────────────────────────────────────────────────
# prometheus_client: Exposes /metrics endpoint in Prometheus text format
# This is what the enterprise monitoring stack scrapes for HPA decisions
prometheus_client==0.21.0

# ── Pydantic ──────────────────────────────────────────────────────────────────
# FastAPI uses Pydantic for request/response validation
# Pydantic v2 (2.x) is significantly faster than v1 — Rust-based core
pydantic==2.9.2

# ── Production utilities ──────────────────────────────────────────────────────
# python-multipart: Required for FastAPI form data handling
python-multipart==0.0.12

# httpx: Async HTTP client (useful for health checks calling other services,
# or downloading model artifacts in init container scripts)
httpx==0.27.2
